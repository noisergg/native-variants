## Quick Start

```tsx
import { createNVA, ThemeProvider, useTheme } from "native-variants";

// 1. Create your theme
const { styled, theme, colorScheme } = createNVA({
  theme: {
    colors: {
      default: {
        primary: "#3b82f6",
        background: "#ffffff",
        foreground: "#000000",
      },
      dark: {
        primary: "#60a5fa",
        background: "#0a0a0a",
        foreground: "#ffffff",
      },
    },
  },
});

// 2. Create styled variants
const buttonVariants = styled((ctx, t) => ctx({
  slots: ["root", "text"],
  base: {
    root: {
      paddingHorizontal: t.spacing["4"],
      paddingVertical: t.spacing["3"],
      borderRadius: t.radii.lg,
    },
    text: {
      fontSize: t.fontSizes.base,
      fontWeight: t.fontWeights.semibold,
    },
  },
  variants: {
    variant: {
      solid: {
        root: { backgroundColor: t.colors.primary },
        text: { color: t.colors.background },
      },
      outline: {
        root: { 
          backgroundColor: "transparent",
          borderWidth: t.borderWidths.DEFAULT,
          borderColor: t.colors.primary,
        },
        text: { color: t.colors.primary },
      },
    },
  },
  defaultVariants: {
    variant: "solid",
  },
}));

// 3. Wrap your app with ThemeProvider
function App() {
  return (
    <ThemeProvider colors={colorScheme}>
      <MyApp />
    </ThemeProvider>
  );
}

// 4. Use in components
function Button({ variant, children }) {
  const { root, text } = buttonVariants({ variant });
  const { colors } = useTheme();

  return (
    <TouchableOpacity style={root}>
      <Text style={text}>{children}</Text>
    </TouchableOpacity>
  );
}
```

## Core Features

### Variants

Create multiple versions of the same component with a clean, declarative API.

```tsx
const buttonVariants = styled((ctx, t) => ctx({
  slots: ["root", "text"],
  base: {
    root: { paddingHorizontal: 16, paddingVertical: 12 },
    text: { color: "#ffffff", textAlign: "center" },
  },
  variants: {
    variant: {
      solid: {
        root: { backgroundColor: t.colors.primary },
      },
      ghost: {
        root: { backgroundColor: "transparent" },
        text: { color: t.colors.primary },
      },
    },
    size: {
      sm: { root: { paddingVertical: 8 } },
      lg: { root: { paddingVertical: 16 } },
    },
    disabled: {
      true: { root: { opacity: 0.5 } },
      false: { root: { opacity: 1 } },
    },
  },
  defaultVariants: {
    variant: "solid",
    size: "sm",
    disabled: false,
  },
}));

// Usage
const styles = buttonVariants({ variant: "ghost", size: "lg" });
```

### Compound Variants

Apply styles based on multiple variant combinations.

```tsx
const buttonVariants = styled((ctx, t) => ctx({
  // ...base and variants
  compoundVariants: [
    {
      variant: "solid",
      disabled: true,
      css: {
        root: { backgroundColor: t.palette.gray400 },
        text: { color: t.palette.gray600 },
      },
    },
  ],
}));
```

### Slots

Split components into multiple parts, each with their own styles.

```tsx
const cardVariants = styled((ctx, t) => ctx({
  slots: ["root", "header", "content", "footer"],
  base: {
    root: {
      backgroundColor: t.colors.background,
      borderRadius: t.radii.lg,
      ...t.shadows.md,
    },
    header: {
      padding: t.spacing["4"],
      borderBottomWidth: 1,
    },
    content: {
      padding: t.spacing["4"],
    },
    footer: {
      padding: t.spacing["4"],
      borderTopWidth: 1,
    },
  },
}));
```

### Built-in Tailwind CSS Tokens

Access Tailwind CSS design tokens directly in your styles.

```tsx
const { styled, theme } = createNVA({ theme: { colors: { /* ... */ } } });

// Use tokens in your styles
const styles = {
  container: {
    padding: theme.spacing["4"],        // 16
    borderRadius: theme.radii.lg,       // 8
    ...theme.shadows.md,                // shadow object
  },
  text: {
    fontSize: theme.fontSizes.lg,       // 18
    fontWeight: theme.fontWeights.bold, // "700"
    letterSpacing: theme.letterSpacing.tight, // -0.4
  },
};

// Full Tailwind color palette
const blue = theme.palette.blue500;     // "#3b82f6"
const red = theme.palette.red500;       // "#ef4444"
```

### Dark/Light Theme Support

Built-in theme provider with system preference detection and persistence.

```tsx
import { ThemeProvider, useTheme, useIsDark } from "native-variants";

function App() {
  return (
    <ThemeProvider 
      colors={colorScheme}
      defaultMode="system"
    >
      <MyApp />
    </ThemeProvider>
  );
}

function MyComponent() {
  const { colors, isDark, toggle, setMode } = useTheme();

  return (
    <View style={{ backgroundColor: colors.background }}>
      <Text style={{ color: colors.foreground }}>
        {isDark ? "Dark Mode" : "Light Mode"}
      </Text>
      <Button onPress={toggle} title="Toggle Theme" />
    </View>
  );
}
```

### Style Merging with `cn`

Merge styles easily, compatible with `react-native-reanimated`.

```tsx
import { cn } from "native-variants";

const style = cn(
  styles.root,
  animatedStyle,
  { padding: 16 },
  isActive && styles.active
);
```

### Color Utilities

Manipulate colors with built-in utilities.

```tsx
import { alpha, lighten, darken } from "native-variants";

const color = alpha(theme.colors.primary, 0.5);  // 50% opacity
const lighter = lighten(theme.colors.primary, 0.2);
const darker = darken(theme.colors.primary, 0.2);
```

## Developer Experience

Native Variants was built with the developer experience in mind:

- **Full TypeScript support** with autocomplete for variants, slots, and tokens
- **Type-safe color schemes** - TypeScript ensures `dark` and `default` colors have the same keys
- **Performance optimized** with built-in caching for variant computations
- **JSDoc documentation** with inline examples and token values
